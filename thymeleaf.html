thymeleaf

Created Wednesday 23 March 2022

thymeleaf is a template engine.
To add thymeleaf in spring project use the following dependencies 

		<dependency>
			<groupId>org.springframework.boot</groupId>
			<artifactId>spring-boot-starter-thymeleaf</artifactId>
		</dependency>

now running the spring-boot project gives the following warning

Cannot find template location: classpath:/templates/ (please add some templates or check your Thymeleaf configuration)

which simply means there were no template file found at the default template location

add the template file home.html in /template/ folder 

<!DOCTYPE html>
<!-- xmlns means xml namespace -->
<html lang="en" xmlns:th="http://www.thymeleaf.org">

<head>
	<title>Demo Thymeleaf</title>
</head>

<body>
	<div align="center">
		<h1> This is a demo of Thymleaf </h1>
		<!-- th is thymeleaf prefix -->
		<!-- text is use to set the text for the body of html tag -->
		<!-- so Play is displayed instead of click here to google-->
		    <!-- <a th:text="Play" href="https://www.google.com">Click here to google</a> -->
		<!-- so below can be used instead -->
		<a th:href="play" th:text="Play"></a>
	</div>
</body>
</html>

also we create ViewNames.java final class to store a string constant HOME = "home". later we can use this constant name to map it to the "/" root view using spring 

package com.blogspot.nodemorph.thymeleafdemo.util;

public final class ViewNames {
	
	//constant
	public static final String HOME = "home";
	
	//private constructor to prevent class instantiation
	private ViewNames( ) {
		
	}
}

now create WebConfig.java class to set view name 

package com.blogspot.nodemorph.thymeleafdemo.config;

import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.config.annotation.ViewControllerRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

import com.blogspot.nodemorph.thymeleafdemo.util.ViewNames;

@Configuration
public class WebConfig implements WebMvcConfigurer {

	@Override
	public void addViewControllers(ViewControllerRegistry registry) {
		registry.addViewController("/").setViewName(ViewNames.HOME);  // <-- here we set the viewcontroller "/" to home.html
		// registry.addViewController("home").setViewName(ViewNames.HOME); <-- another example
	}
}
